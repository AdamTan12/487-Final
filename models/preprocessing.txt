# Classifier preprocessing contract.
#
# Source of truth for the values both the Python smoke test and the C++
# inference path must use. Any drift here silently degrades accuracy --
# update both sides in lockstep.
#
# Values are HaGRID's own training preprocessing (see
# https://github.com/hukenovs/hagrid/blob/master/configs/ResNet18.yaml).
# We did NOT train the classifier ourselves -- we use HaGRID's published
# ResNet18 checkpoint via training/export_onnx.py.

input_size:  224x224
color_order: RGB                          # cv::dnn blob built with swapRB=true
pad_color:   [144, 144, 144]              # gray letterbox padding (BGR)
mean:        [0.54, 0.499, 0.474]         # in [0,1] units (post-scale)
std:         [0.234, 0.235, 0.231]
scale:       0.00392156862745098          # 1.0 / 255.0

# 34 HaGRID classes (33 gestures + no_gesture). The classifier head emits
# logits of length 34 in this exact order.
class_order: [
    grabbing, grip, holy, point, call, three3, timeout, xsign,
    hand_heart, hand_heart2, little_finger, middle_finger, take_picture,
    dislike, fist, four, like, mute, ok, one, palm, peace,
    peace_inverted, rock, stop, stop_inverted, three, three2,
    two_up, two_up_inverted, three_gun, thumb_index, thumb_index2,
    no_gesture
]

# Letterbox protocol: resize so longest side == input_size, pad the
# remaining short-side margin equally with pad_color, then normalize.
# This matches HaGRID's Albumentations LongestMaxSize + PadIfNeeded(value=144)
# pipeline byte-for-byte.
